#include "Particle.h"
#include "../lib/neopixel/src/neopixel.h"
#include "Adafruit_GFX.h"
#include "Adafruit_ST7735.h"
#include "Adafruit_Seesaw.h"

SYSTEM_MODE(AUTOMATIC);
SYSTEM_THREAD(ENABLED);
SerialLogHandler logHandler(LOG_LEVEL_INFO);

// ── NeoPixel ──────────────────────────────────────────────────────────────────
#define PIXEL_COUNT 12
#define PIXEL_PIN   SPI1    // = D2 on your board
#define PIXEL_TYPE  WS2812
Adafruit_NeoPixel ring(PIXEL_COUNT, PIXEL_PIN, PIXEL_TYPE);

// ── TFT Display ───────────────────────────────────────────────────────────────
// Your wiring: TCS=A5, D/C=A1, RST=A2
#define TFT_CS  S3
#define TFT_DC  A1
#define TFT_RST A2
Adafruit_ST7735 display = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_RST);

// Colours
#define COLOR_BLACK   0x0000
#define COLOR_WHITE   0xFFFF
#define COLOR_GREEN   0x07E0
#define COLOR_YELLOW  0xFFE0
#define COLOR_RED     0xF800

// ── Soil Sensor ───────────────────────────────────────────────────────────────
Adafruit_Seesaw soilSensor;
bool soilSensorFound = false;
bool soilReadError   = false;

// ── Plant state ───────────────────────────────────────────────────────────────
int plantState     = 0;
int lastPlantState = -1;

// Moisture thresholds (200=very dry, 2000=very wet)
#define THRESHOLD_HEALTHY  800
#define THRESHOLD_WARNING  400

#define READ_INTERVAL_MS 5000
unsigned long lastReadTime = 0;

// ── NeoPixel colour helper ────────────────────────────────────────────────────
void setRingColor(uint8_t r, uint8_t g, uint8_t b) {
    for (int i = 0; i < ring.numPixels(); i++) {
        ring.setPixelColor(i, r, g, b);
    }
    ring.setBrightness(60);
    ring.show();
}

// ── Draw face on TFT ─────────────────────────────────────────────────────────
void drawFace(int faceType, int faceColor) {
    display.fillScreen(COLOR_BLACK);
    display.drawCircle(64, 52, 34, faceColor);
    display.fillCircle(52, 42, 4, faceColor);
    display.fillCircle(76, 42, 4, faceColor);

    if (faceType == 0) {
        // Happy
        for (int x = -12; x <= 12; x++) {
            int y = 4 - (x * x) / 16;
            display.drawPixel(64 + x, 66 + y, faceColor);
            display.drawPixel(64 + x, 67 + y, faceColor);
        }
    } else if (faceType == 1) {
        // Neutral
        display.drawFastHLine(52, 66, 24, faceColor);
        display.drawFastHLine(52, 67, 24, faceColor);
    } else {
        // Sad
        for (int x = -12; x <= 12; x++) {
            int y = (x * x) / 16;
            display.drawPixel(64 + x, 66 + y, faceColor);
            display.drawPixel(64 + x, 67 + y, faceColor);
        }
    }
}

// ── Draw status text ──────────────────────────────────────────────────────────
void drawStatusText(int state) {
    display.setTextSize(1);
    display.setTextWrap(true);
    if (state == 0) {
        display.setTextColor(COLOR_GREEN);
        display.setCursor(18, 96);
        display.print("Plant is happy!");
        display.setCursor(10, 110);
        display.print("No action needed");
    } else if (state == 1) {
        display.setTextColor(COLOR_YELLOW);
        display.setCursor(8, 96);
        display.print("Needs water soon");
        display.setCursor(8, 110);
        display.print("Check soil today");
    } else {
        display.setTextColor(COLOR_RED);
        display.setCursor(8, 96);
        display.print("Needs water NOW");
        display.setCursor(8, 110);
        display.print("Soil very dry!");
    }
}

// ── Update NeoPixel + TFT ────────────────────────────────────────────────────
void updateOutputs(int state) {
    if (state == 0) {
        setRingColor(0, 180, 0);
        drawFace(0, COLOR_GREEN);
        drawStatusText(0);
        Log.info("State: HEALTHY");
    } else if (state == 1) {
        setRingColor(180, 140, 0);
        drawFace(1, COLOR_YELLOW);
        drawStatusText(1);
        Log.info("State: WARNING");
    } else {
        setRingColor(200, 0, 0);
        drawFace(2, COLOR_RED);
        drawStatusText(2);
        Log.info("State: CRITICAL");
    }
}

void showSensorError() {
    setRingColor(180, 0, 180);
    display.fillScreen(COLOR_BLACK);
    drawFace(2, COLOR_RED);
    display.setTextSize(1);
    display.setTextWrap(true);
    display.setTextColor(COLOR_RED);
    display.setCursor(10, 96);
    display.print("Sensor read error");
    display.setTextColor(COLOR_WHITE);
    display.setCursor(6, 110);
    display.print("Check I2C / soil");
    Log.error("Soil sensor returned an invalid reading.");
}

void showSensorNotFound() {
    setRingColor(180, 0, 180);
    display.fillScreen(COLOR_BLACK);
    drawFace(2, COLOR_RED);
    display.setTextSize(1);
    display.setTextWrap(true);
    display.setTextColor(COLOR_RED);
    display.setCursor(16, 96);
    display.print("Sensor missing");
    display.setTextColor(COLOR_WHITE);
    display.setCursor(6, 110);
    display.print("Check D0/D1 power");
}

// ── setup() ──────────────────────────────────────────────────────────────────
void setup() {
    waitFor(Serial.isConnected, 5000);
    Log.info("Booting Plant Care Companion firmware");

    // NeoPixel — dim blue on boot
    ring.begin();
    setRingColor(0, 0, 40);

    // FIX 1: explicitly start I2C before soil sensor
    delay(500);
    Wire.setSpeed(CLOCK_SPEED_100KHZ);
    Wire.begin();
    delay(500);

    // FIX 2: INITR_144GREENTAB is the correct init for the 1.44" display
    display.initR(INITR_GREENTAB);
    display.fillScreen(COLOR_BLACK);
    display.setTextColor(COLOR_WHITE);
    display.setTextSize(1);
    display.setCursor(10, 50);
    display.print("Starting up...");

    // Start soil sensor
    if (!soilSensor.begin(0x36)) {
        Log.error("Soil sensor not found! Check D0/D1 wiring.");
        soilSensorFound = false;
        display.fillScreen(COLOR_BLACK);
        display.setTextColor(COLOR_RED);
        display.setCursor(5, 40);
        display.print("Soil sensor");
        display.setCursor(5, 54);
        display.print("not found.");
        display.setCursor(5, 70);
        display.setTextColor(COLOR_WHITE);
        display.print("Check D0+D1");
    } else {
        Log.info("Soil sensor ready at I2C address 0x36.");
        soilSensorFound = true;
        display.fillScreen(COLOR_BLACK);
        display.setTextColor(COLOR_GREEN);
        display.setCursor(14, 50);
        display.print("Sensor ready!");
    }

    delay(1500);
}

// ── loop() ───────────────────────────────────────────────────────────────────
void loop() {
    unsigned long now = millis();

    if (now - lastReadTime >= READ_INTERVAL_MS) {
        lastReadTime = now;
        soilReadError = false;

        if (soilSensorFound) {
            uint16_t moisture = soilSensor.touchRead(0);
            float    tempC    = soilSensor.getTemp();
            Log.info("Moisture: %d   Temp: %.1f C", moisture, tempC);

            soilReadError = (moisture == 0xFFFF);

            if (soilReadError) {
                Log.error("Moisture read failed (0xFFFF). Check SDA=D0, SCL=D1, power, and I2C speed.");
            } else if (moisture >= THRESHOLD_HEALTHY) {
                plantState = 0;
            } else if (moisture >= THRESHOLD_WARNING) {
                plantState = 1;
            } else {
                plantState = 2;
            }
        } else {
            Log.error("Skipping read because soil sensor was not detected during setup.");
            showSensorNotFound();
            lastPlantState = -1;
        }

        if (soilReadError) {
            showSensorError();
            lastPlantState = -1;
        } else if (soilSensorFound && plantState != lastPlantState) {
            updateOutputs(plantState);
            lastPlantState = plantState;
        }

        Log.info("Loop summary -> sensorFound=%d, readError=%d, state=%d", soilSensorFound, soilReadError, plantState);
    }
}
